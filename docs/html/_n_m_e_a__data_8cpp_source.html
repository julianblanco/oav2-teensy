50void Adafruit_GPS::newDataValue(nmea_index_t idx, nmea_float_t v) {
51#ifdef NMEA_EXTENSIONS
52 // Serial.println();Serial.print(idx);Serial.print(", "); Serial.println(v);
53 val[idx].latest = v; // update the value
54
55 // update the smoothed verion
56 if (isCompoundAngle(idx)) { // angle with sin/cos component recording
57 newDataValue((nmea_index_t)(idx + 1), sin(v / (nmea_float_t)RAD_TO_DEG));
58 newDataValue((nmea_index_t)(idx + 2), cos(v / (nmea_float_t)RAD_TO_DEG));
59 }
60 // weighting factor for smoothing depends on delta t / tau
61 nmea_float_t w =
62 min((nmea_float_t)1.0,
63 (nmea_float_t)(millis() - val[idx].lastUpdate) / val[idx].response);
64 // default smoothing
65 val[idx].smoothed = (1.0f - w) * val[idx].smoothed + w * v;
66 // special smoothing for some angle types
67 if (val[idx].type == NMEA_COMPASS_ANGLE_SIN)
68 val[idx].smoothed =
69 compassAngle(val[idx + 1].smoothed, val[idx + 2].smoothed);
70 if (val[idx].type == NMEA_BOAT_ANGLE_SIN)
71 val[idx].smoothed = boatAngle(val[idx + 1].smoothed, val[idx + 2].smoothed);
72 // some types just don't make sense to smooth -- use latest
73 if (val[idx].type == NMEA_BOAT_ANGLE)
74 val[idx].smoothed = val[idx].latest;
75 if (val[idx].type == NMEA_COMPASS_ANGLE)
76 val[idx].smoothed = val[idx].latest;
77 if (val[idx].type == NMEA_DDMM)
78 val[idx].smoothed = val[idx].latest;
79 if (val[idx].type == NMEA_HHMMSS)
80 val[idx].smoothed = val[idx].latest;
81
82 val[idx].lastUpdate = millis(); // take a time stamp
83 if (val[idx].hist) { // there's a history struct for this tag
84 unsigned long seconds = (millis() - val[idx].hist->lastHistory) / 1000;
85 // do an update if the time has come, or if this is the first time through
86 if (seconds >= val[idx].hist->historyInterval ||
87 val[idx].hist->lastHistory == 0) {
88
89 // move the old history back in time by one step
90 for (unsigned i = 0; i < (val[idx].hist->n - 1); i++)
91 val[idx].hist->data[i] = val[idx].hist->data[i + 1];
92
93 // Create the new entry, scaling and offsetting the value to fit into an
94 // integer, and based on the smoothed value.
95 val[idx].hist->data[val[idx].hist->n - 1] =
96 val[idx].hist->scale * (val[idx].smoothed - val[idx].hist->offset);
97 val[idx].hist->lastHistory = millis();
98 }
99 }
100#endif // NMEA_EXTENSIONS
101}

383nmea_history_t *Adafruit_GPS::initHistory(nmea_index_t idx, nmea_float_t scale,
384 nmea_float_t offset,
385 unsigned historyInterval,
386 unsigned historyN) {
387 historyN = max((unsigned)10, historyN);
388 if (idx < NMEA_MAX_INDEX) {
389 // remove any existing history
390 if (val[idx].hist != NULL)
391 removeHistory(idx);
392 // space for the struct
393 val[idx].hist = (nmea_history_t *)malloc(sizeof(nmea_history_t));
394 if (val[idx].hist != NULL) {
395 // space for the data array of the appropriate size
396 val[idx].hist->data = (int16_t *)malloc(sizeof(int16_t) * historyN);
397 if (val[idx].hist->data != NULL) {
398 // initialize the data array
399 for (unsigned i = 0; i < historyN; i++)
400 val[idx].hist->data[i] = 0;
401 } else
402 free(val[idx].hist);
403 }
404 if (val[idx].hist != NULL) {
405 val[idx].hist->n = historyN;
406 if (scale > 0.0f)
407 val[idx].hist->scale = scale;
408 val[idx].hist->offset = offset;
409 if (historyInterval > 0)
410 val[idx].hist->historyInterval = historyInterval;
411 }
412 return val[idx].hist;
413 }
414 return NULL;
415}

519nmea_float_t Adafruit_GPS::boatAngle(nmea_float_t s, nmea_float_t c) {
520 // put the sin angle in -90 to 90 range
521 nmea_float_t sAng = asin(s) * (nmea_float_t)RAD_TO_DEG;
522 while (sAng < -90)
523 sAng += 180.0f;
524 while (sAng > 90)
525 sAng -= 180.0f;
526 // put the cos angle in 0 to 180 range
527 nmea_float_t cAng = acos(c) * (nmea_float_t)RAD_TO_DEG;
528 while (cAng < 0)
529 cAng += 180.0f;
530 while (cAng > 180)
531 cAng -= 180.0f;
532 // Pick the most accurate representation and translate
533 if (cAng < 45)
534 return sAng; // Close hauled
535 else {
536 if (cAng > 135) { // Running
537 if (sAng > 0)
538 return 180 - sAng; // on starboard tack
539 else
540 return -180 - sAng; // on port tack
541 } else { // Reaching
542 if (sAng < 0)
543 return -cAng; // on port tack
544 else
545 return cAng; // on starboard tack
546 }
547 }
548 return 9999; // you can't get here, but there must be an explicit return
549}

560nmea_float_t Adafruit_GPS::compassAngle(nmea_float_t s, nmea_float_t c) {
561 nmea_float_t ang = boatAngle(s, c);
562 if (ang < 5000) { // if reasonable range
563 while (ang < 0)
564 ang += 360.0f; // round up
565 while (ang > 360)
566 ang -= 360.0f; // round down
567 }
568 return ang;
