32/*------------------------------------------------------------------------------
33 Configure
34
35 Selects one of several preset configurations.
36
37 Parameters
38 ------------------------------------------------------------------------------
39 configuration: Default 0.
40 0: Maximum range. Uses maximum acquisition count.
41 1: Balanced performance.
42 2: Short range, high speed. Reduces maximum acquisition count.
43 3: Mid range, higher speed. Turns on quick termination
44 detection for faster measurements at short range (with decreased
45 accuracy)
46 4: Maximum range, higher speed on short range targets. Turns on quick
47 termination detection for faster measurements at short range (with
48 decreased accuracy)
49 5: Very short range, higher speed, high error. Reduces maximum
50 acquisition count to a minimum for faster rep rates on very
51 close targets with high error.
52 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
53 operating manual for instructions.
54------------------------------------------------------------------------------*/
55void LIDARLite_v4LED::configure(uint8_t configuration, uint8_t lidarliteAddress)
56{
57 uint8_t sigCountMax;
58 uint8_t acqConfigReg;
59
60 switch (configuration)
61 {
62 case 0: // Default mode - Maximum range
63 sigCountMax = 0xff;
64 acqConfigReg = 0x08;
65 break;
66
67 case 1: // Balanced performance
68 sigCountMax = 0x80;
69 acqConfigReg = 0x08;
70 break;
71
72 case 2: // Short range, high speed
73 sigCountMax = 0x18;
74 acqConfigReg = 0x00;
75 break;
76
77 case 3: // Mid range, higher speed on short range targets
78 sigCountMax = 0x80;
79 acqConfigReg = 0x00;
80 break;
81
82 case 4: // Maximum range, higher speed on short range targets
83 sigCountMax = 0xff;
84 acqConfigReg = 0x00;
85 break;
86
87 case 5: // Very short range, higher speed, high error
88 sigCountMax = 0x04;
89 acqConfigReg = 0x00;
90 break;
91 }
92
93 write(0x05, &sigCountMax , 1, lidarliteAddress);
94 write(0xE5, &acqConfigReg , 1, lidarliteAddress);
95} /* LIDARLite_v4LED::configure */

97/*------------------------------------------------------------------------------
98 Set I2C Address
99
100 Set Alternate I2C Device Address. See Operation Manual for additional info.
101
102 Parameters
103 ------------------------------------------------------------------------------
104 newAddress: desired secondary I2C device address
105 disableDefault: a non-zero value here means the default 0x62 I2C device
106 address will be disabled.
107 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
108 operating manual for instructions.
109------------------------------------------------------------------------------*/
110void LIDARLite_v4LED::setI2Caddr (uint8_t newAddress, uint8_t disableDefault,
111 uint8_t lidarliteAddress)
112{
113 uint8_t dataBytes[5];
114
115 // Read 4-byte device serial number
116 read (0x16, dataBytes, 4, lidarliteAddress);
117
118 // Append the desired I2C address to the end of the serial number byte array
119 dataBytes[4] = newAddress;
120
121 // Write the serial number and new address in one 5-byte transaction
122 write(0x16, dataBytes, 5, lidarliteAddress);
123
124 // If desired, disable default I2C device address (using the new I2C device address)
125 if (disableDefault)
126 {
127 dataBytes[0] = 0x01; // set bit to disable default address
128 write(0x1b, dataBytes, 1, newAddress);
129 }
130} /* LIDARLite_v4LED::setI2Caddr */

170/*------------------------------------------------------------------------------
171 Get Busy Flag
172
173 Read BUSY flag from device registers. Function will return 0x00 if not busy.
174
175 Parameters
176 ------------------------------------------------------------------------------
177 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
178 operating manual for instructions.
179------------------------------------------------------------------------------*/
180uint8_t LIDARLite_v4LED::getBusyFlag(uint8_t lidarliteAddress)
181{
182 uint8_t statusByte = 0;
183 uint8_t busyFlag; // busyFlag monitors when the device is done with a measurement
184
185 // Read status register to check busy flag
186 read(0x01, &statusByte, 1, lidarliteAddress);
187
188 // STATUS bit 0 is busyFlag
189 busyFlag = statusByte & 0x01;
190
191 return busyFlag;
192} /* LIDARLite_v4LED::getBusyFlag */

195 Take Range using Trigger / Monitor Pins
196
197 Initiate a distance measurement by toggling the trigger pin
198
199 Parameters
200 ------------------------------------------------------------------------------
201 triggerPin: digital output pin connected to trigger input of LIDAR-Lite
202 monitorPin: digital input pin connected to monitor output of LIDAR-Lite
203------------------------------------------------------------------------------*/
204void LIDARLite_v4LED::takeRangeGpio(uint8_t triggerPin, uint8_t monitorPin)
205{
206 uint8_t busyFlag;
207
208 if (digitalRead(triggerPin))
209 digitalWrite(triggerPin, LOW);
210 else
211 digitalWrite(triggerPin, HIGH);
212
213 // When LLv4 receives trigger command it will drive monitor pin low.
214 // Wait for LLv4 to acknowledge receipt of command before moving on.
215 do
216 {
217 busyFlag = getBusyFlagGpio(monitorPin);
218 } while (!busyFlag);
219} /* LIDARLite_v4LED::takeRangeGpio */

284/*------------------------------------------------------------------------------
285 Write
286
287 Perform I2C write to device. The I2C peripheral in the LidarLite v3 HP
288 will receive multiple bytes in one I2C transmission. The first byte is
289 always the register address. The the bytes that follow will be written
290 into the specified register address first and then the internal address
291 in the Lidar Lite will be auto-incremented for all following bytes.
292
293 Parameters
294 ------------------------------------------------------------------------------
295 regAddr: register address to write to
296 dataBytes: pointer to array of bytes to write
297 numBytes: number of bytes in 'dataBytes' array to write
298 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
299 operating manual for instructions.
300------------------------------------------------------------------------------*/
301void LIDARLite_v4LED::write(uint8_t regAddr, uint8_t * dataBytes,
302 uint8_t numBytes, uint8_t lidarliteAddress)
303{
304 int nackCatcher;
305
306 Wire.beginTransmission((int) lidarliteAddress);
307
308 // First byte of every write sets the LidarLite's internal register address pointer
309 Wire.write((int) regAddr);
310
311 // Subsequent bytes are data writes
312 Wire.write(dataBytes, (int) numBytes);
313
314 // A nack means the device is not responding. Report the error over serial.
315 nackCatcher = Wire.endTransmission();
316 if (nackCatcher != 0)
317 {
318 // handle nack issues in here
319 }
320} /* LIDARLite_v4LED::write */

322/*------------------------------------------------------------------------------
323 Read
324
325 Perform I2C read from device. The I2C peripheral in the LidarLite v3 HP
326 will send multiple bytes in one I2C transmission. The register address must
327 be set up by a previous I2C write. The bytes that follow will be read
328 from the specified register address first and then the internal address
329 pointer in the Lidar Lite will be auto-incremented for following bytes.
330
331 Will detect an unresponsive device and report the error over serial.
332
333 Parameters
334 ------------------------------------------------------------------------------
335 regAddr: register address to write to
336 dataBytes: pointer to array of bytes to write
337 numBytes: number of bytes in 'dataBytes' array to write
338 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
339 operating manual for instructions.
340------------------------------------------------------------------------------*/
341void LIDARLite_v4LED::read(uint8_t regAddr, uint8_t * dataBytes,
342 uint8_t numBytes, uint8_t lidarliteAddress)
343{
344 uint16_t i = 0;
345 int nackCatcher = 0;
346
347 // Set the internal register address pointer in the Lidar Lite
348 Wire.beginTransmission((int) lidarliteAddress);
349 Wire.write((int) regAddr); // Set the register to be read
350
351 // A nack means the device is not responding, report the error over serial
352 nackCatcher = Wire.endTransmission(false); // false means perform repeated start
353 if (nackCatcher != 0)
354 {
355 // handle nack issues in here
356 }
357
358 // Perform read, save in dataBytes array
359 Wire.requestFrom((int)lidarliteAddress, (int) numBytes);
360 if ((int) numBytes <= Wire.available())
361 {
362 while (i < numBytes)
363 {
364 dataBytes[i] = (uint8_t) Wire.read();
365 i++;
366 }
367 }
368
369} /* LIDARLite_v4LED::read */

371/*------------------------------------------------------------------------------
372 Correlation Record Read
373
374 The correlation record used to calculate distance can be read from the device.
375 It has a bipolar wave shape, transitioning from a positive going portion to a
376 roughly symmetrical negative going pulse. The point where the signal crosses
377 zero represents the effective delay for the reference and return signals.
378
379 Process
380 ------------------------------------------------------------------------------
381 1. Take a distance reading (there is no correlation record without at least
382 one distance reading being taken)
383 2. For as many points as you want to read from the record (max is 192) read
384 the two byte signed correlation data point from 0x52
385
386 Parameters
387 ------------------------------------------------------------------------------
388 correlationArray: pointer to memory location to store the correlation record
389 ** Two bytes for every correlation value must be
390 allocated by calling function
391 numberOfReadings: Default = 192. Maximum = 192
392 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
393 operating manual for instructions.
394------------------------------------------------------------------------------*/
395void LIDARLite_v4LED::correlationRecordRead(
396 int16_t * correlationArray, uint8_t numberOfReadings, uint8_t lidarliteAddress)
397{
398 uint8_t i;
399 int16_t correlationValue;
400 uint8_t * dataBytes = (uint8_t *) &correlationValue;
401
402 for (i=0 ; i<numberOfReadings ; i++)
403 {
404 read(0x52, dataBytes, 2, lidarliteAddress);
405 correlationArray[i] = correlationValue;
406 }
407} /* LIDARLite_v4LED::correlationRecordRead */
