3void loopRate(int freq,float functionstarttime)
4{
5 //DESCRIPTION: Regulate main loop rate to specified frequency in Hz
6 /*
7 * It's good to operate at a constant loop rate for filters to remain stable and whatnot. Interrupt routines running in the
8 * background cause the loop rate to fluctuate. This function basically just waits at the end of every loop iteration until
9 * the correct time has passed since the start of the current loop for the desired loop rate in Hz. 2kHz is a good rate to
10 * be at because the loop nominally will run between 2.8kHz - 4.2kHz. This lets us have a little room to add extra computations
11 * and remain above 2kHz, without needing to retune all of our filtering parameters scattered around this code.
12 */
13 float invFreq = 1.0 / freq * 1000000.0;
14 unsigned long checker = micros();
15
16 //Sit in loop until appropriate time has passed
17 while (invFreq > (checker - functionstarttime))
18 {
19 checker = micros();
20 }
21}
