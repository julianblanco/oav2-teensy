148bool Adafruit_GPS::begin(uint32_t baud_or_i2caddr) {
149#if (defined(__AVR__) || defined(ESP8266)) && defined(USE_SW_SERIAL)
150 if (gpsSwSerial) {
151 gpsSwSerial->begin(baud_or_i2caddr);
152 }
153#endif
154 if (gpsHwSerial) {
155 gpsHwSerial->begin(baud_or_i2caddr);
156 }
157 if (gpsI2C) {
158 gpsI2C->begin();
159 if (baud_or_i2caddr > 0x7F) {
160 _i2caddr = GPS_DEFAULT_I2C_ADDR;
161 } else {
162 _i2caddr = baud_or_i2caddr;
163 }
164 // A basic scanner, see if it ACK's
165 gpsI2C->beginTransmission(_i2caddr);
166 return (gpsI2C->endTransmission() == 0);
167 }
168 if (gpsSPI) {
169 gpsSPI->begin();
170 gpsSPI_settings = SPISettings(baud_or_i2caddr, MSBFIRST, SPI_MODE0);
171 if (gpsSPI_cs >= 0) {
172 pinMode(gpsSPI_cs, OUTPUT);
173 digitalWrite(gpsSPI_cs, HIGH);
174 }
175 }
176
177 delay(10);
178 return true;
179}

362char Adafruit_GPS::read(void) {
363 static uint32_t firstChar = 0; // first character received in current sentence
364 uint32_t tStart = millis(); // as close as we can get to time char was sent
365 char c = 0;
366
367 if (paused || noComms)
368 return c;
369
370#if (defined(__AVR__) || defined(ESP8266)) && defined(USE_SW_SERIAL)
371 if (gpsSwSerial) {
372 if (!gpsSwSerial->available())
373 return c;
374 c = gpsSwSerial->read();
375 }
376#endif
377 if (gpsHwSerial) {
378 if (!gpsHwSerial->available())
379 return c;
380 c = gpsHwSerial->read();
381 }
382 if (gpsI2C) {
383 if (_buff_idx <= _buff_max) {
384 c = _i2cbuffer[_buff_idx];
385 _buff_idx++;
386 } else {
387 // refill the buffer!
388 if (gpsI2C->requestFrom((uint8_t)0x10, (uint8_t)GPS_MAX_I2C_TRANSFER,
389 (uint8_t) true) == GPS_MAX_I2C_TRANSFER) {
390 // got data!
391 _buff_max = 0;
392 char curr_char = 0;
393 for (int i = 0; i < GPS_MAX_I2C_TRANSFER; i++) {
394 curr_char = gpsI2C->read();
395 if ((curr_char == 0x0A) && (last_char != 0x0D)) {
396 // skip duplicate 0x0A's - but keep as part of a CRLF
397 continue;
398 }
399 last_char = curr_char;
400 _i2cbuffer[_buff_max] = curr_char;
401 _buff_max++;
402 }
403 _buff_max--; // back up to the last valid slot
404 if ((_buff_max == 0) && (_i2cbuffer[0] == 0x0A)) {
405 _buff_max = -1; // ahh there was nothing to read after all
406 }
407 _buff_idx = 0;
408 }
409 return c;
410 }
411 }
412
413 if (gpsSPI) {
414 do {
415 gpsSPI->beginTransaction(gpsSPI_settings);
416 if (gpsSPI_cs >= 0) {
417 digitalWrite(gpsSPI_cs, LOW);
418 }
419 c = gpsSPI->transfer(0xFF);
420 if (gpsSPI_cs >= 0) {
421 digitalWrite(gpsSPI_cs, HIGH);
422 }
423 gpsSPI->endTransaction();
424 // skip duplicate 0x0A's - but keep as part of a CRLF
425 } while (((c == 0x0A) && (last_char != 0x0D)) ||
426 (!isprint(c) && !isspace(c)));
427 last_char = c;
428 }
429 // Serial.print(c);
430
431 currentline[lineidx++] = c;
432 if (lineidx >= MAXLINELENGTH)
433 lineidx = MAXLINELENGTH -
434 1; // ensure there is someplace to put the next received character
435
436 if (c == '\n') {
437 currentline[lineidx] = 0;
438
439 if (currentline == line1) {
440 currentline = line2;
441 lastline = line1;
442 } else {
443 currentline = line1;
444 lastline = line2;
445 }
446
447 // Serial.println("----");
448 // Serial.println((char *)lastline);
449 // Serial.println("----");
450 lineidx = 0;
451 recvdflag = true;
452 recvdTime = millis(); // time we got the end of the string
453 sentTime = firstChar;
454 firstChar = 0; // there are no characters yet
455 return c; // wait until next character to set time
456 }
457
458 if (firstChar == 0)
459 firstChar = tStart;
460 return c;
461}

508bool Adafruit_GPS::waitForSentence(const char *wait4me, uint8_t max,
509 bool usingInterrupts) {
510 uint8_t i = 0;
511 while (i < max) {
512 if (!usingInterrupts)
513 read();
514
515 if (newNMEAreceived()) {
516 char *nmea = lastNMEA();
517 i++;
518
519 if (strStartsWith(nmea, wait4me))

521 }

556/**************************************************************************/
557bool Adafruit_GPS::LOCUS_ReadStatus(void) {
558 sendCommand(PMTK_LOCUS_QUERY_STATUS);
559
560 if (!waitForSentence("$PMTKLOG"))
561 return false;
562
563 char *response = lastNMEA();
564 uint16_t parsed[10];
565 uint8_t i;
566
567 for (i = 0; i < 10; i++)
568 parsed[i] = -1;
569
570 response = strchr(response, ',');
571 for (i = 0; i < 10; i++) {
572 if (!response || (response[0] == 0) || (response[0] == '*'))
573 break;
574 response++;
575 parsed[i] = 0;
576 while ((response[0] != ',') && (response[0] != '*') && (response[0] != 0)) {
577 parsed[i] *= 10;
578 char c = response[0];
579 if (isDigit(c))
580 parsed[i] += c - '0';
581 else
582 parsed[i] = c;
583 response++;
584 }
585 }
586 LOCUS_serial = parsed[0];
587 LOCUS_type = parsed[1];
588 if (isAlpha(parsed[2])) {
589 parsed[2] = parsed[2] - 'a' + 10;
590 }
591 LOCUS_mode = parsed[2];
592 LOCUS_config = parsed[3];
593 LOCUS_interval = parsed[4];
594 LOCUS_distance = parsed[5];
595 LOCUS_speed = parsed[6];
596 LOCUS_status = !parsed[7];
597 LOCUS_records = parsed[8];
598 LOCUS_percent = parsed[9];
599
600 return true;
601}
