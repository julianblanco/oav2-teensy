32/*------------------------------------------------------------------------------
33 Configure
34
35 Selects one of several preset configurations.
36
37 Parameters
38 ------------------------------------------------------------------------------
39 configuration: Default 0.
40 0: Default mode, balanced performance.
41 1: Short range, high speed. Uses 0x1d maximum acquisition count.
42 2: Default range, higher speed short range. Turns on quick termination
43 detection for faster measurements at short range (with decreased
44 accuracy)
45 3: Maximum range. Uses 0xff maximum acquisition count.
46 4: High sensitivity detection. Overrides default valid measurement detection
47 algorithm, and uses a threshold value for high sensitivity and noise.
48 5: Low sensitivity detection. Overrides default valid measurement detection
49 algorithm, and uses a threshold value for low sensitivity and noise.
50 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
51 operating manual for instructions.
52------------------------------------------------------------------------------*/
53void LIDARLite_v3HP::configure(uint8_t configuration, uint8_t lidarliteAddress)
54{
55 uint8_t sigCountMax;
56 uint8_t acqConfigReg;
57 uint8_t refCountMax;
58 uint8_t thresholdBypass;
59
60 switch (configuration)
61 {
62 case 0: // Default mode, balanced performance
63 sigCountMax = 0x80; // Default
64 acqConfigReg = 0x08; // Default
65 refCountMax = 0x05; // Default
66 thresholdBypass = 0x00; // Default
67 break;
68
69 case 1: // Short range, high speed
70 sigCountMax = 0x1d;
71 acqConfigReg = 0x08; // Default
72 refCountMax = 0x03;
73 thresholdBypass = 0x00; // Default
74 break;
75
76 case 2: // Default range, higher speed short range
77 sigCountMax = 0x80; // Default
78 acqConfigReg = 0x00;
79 refCountMax = 0x03;
80 thresholdBypass = 0x00; // Default
81 break;
82
83 case 3: // Maximum range
84 sigCountMax = 0xff;
85 acqConfigReg = 0x08; // Default
86 refCountMax = 0x05; // Default
87 thresholdBypass = 0x00; // Default
88 break;
89
90 case 4: // High sensitivity detection, high erroneous measurements
91 sigCountMax = 0x80; // Default
92 acqConfigReg = 0x08; // Default
93 refCountMax = 0x05; // Default
94 thresholdBypass = 0x80;
95 break;
96
97 case 5: // Low sensitivity detection, low erroneous measurements
98 sigCountMax = 0x80; // Default
99 acqConfigReg = 0x08; // Default
100 refCountMax = 0x05; // Default
101 thresholdBypass = 0xb0;
102 break;
103
104 case 6: // Short range, high speed, higher error
105 sigCountMax = 0x04;
106 acqConfigReg = 0x01; // turn off short_sig, mode pin = status output mode
107 refCountMax = 0x03;
108 thresholdBypass = 0x00;
109 break;
110 }
111
112 write(0x02, &sigCountMax , 1, lidarliteAddress);
113 write(0x04, &acqConfigReg , 1, lidarliteAddress);
114 write(0x12, &refCountMax , 1, lidarliteAddress);
115 write(0x1c, &thresholdBypass, 1, lidarliteAddress);
116} /* LIDARLite_v3HP::configure */

118/*------------------------------------------------------------------------------
119 Set I2C Address
120
121 Set Alternate I2C Device Address. See Operation Manual for additional info.
122
123 Parameters
124 ------------------------------------------------------------------------------
125 newAddress: desired secondary I2C device address
126 disableDefault: a non-zero value here means the default 0x62 I2C device
127 address will be disabled.
128 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
129 operating manual for instructions.
130------------------------------------------------------------------------------*/
131void LIDARLite_v3HP::setI2Caddr (uint8_t newAddress, uint8_t disableDefault,
132 uint8_t lidarliteAddress)
133{
134 uint8_t dataBytes[2];
135
136 // Read UNIT_ID serial number bytes and write them into I2C_ID byte locations
137 read (0x16, dataBytes, 2, lidarliteAddress);
138 write(0x18, dataBytes, 2, lidarliteAddress);
139
140 // Write the new I2C device address to registers
141 // left shift by one to work around data alignment issue in v3HP
142 dataBytes[0] = (newAddress << 1);
143 write(0x1a, dataBytes, 1, lidarliteAddress);
144
145 // Enable the new I2C device address using the default I2C device address
146 read (0x1e, dataBytes, 1, lidarliteAddress);
147 dataBytes[0] = dataBytes[0] | (1 << 4); // set bit to enable the new address
148 write(0x1e, dataBytes, 1, lidarliteAddress);
149
150 // If desired, disable default I2C device address (using the new I2C device address)
151 if (disableDefault)
152 {
153 read (0x1e, dataBytes, 1, newAddress);
154 dataBytes[0] = dataBytes[0] | (1 << 3); // set bit to disable default address
155 write(0x1e, dataBytes, 1, newAddress);
156 }
157} /* LIDARLite_v3HP::setI2Caddr */

176/*------------------------------------------------------------------------------
177 Wait for Busy Flag
178
179 Blocking function to wait until the Lidar Lite's internal busy flag goes low
180
181 Parameters
182 ------------------------------------------------------------------------------
183 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
184 operating manual for instructions.
185------------------------------------------------------------------------------*/
186void LIDARLite_v3HP::waitForBusy(uint8_t lidarliteAddress)
187{
188 uint16_t busyCounter = 0; // busyCounter counts number of times busy flag is checked, for timeout
189 uint8_t busyFlag = 1; // busyFlag monitors when the device is done with a measurement
190
191 while (busyFlag) // Loop until device is not busy
192 {
193 // Handle timeout condition, exit while loop and goto bailout
194 if (busyCounter > 9999)
195 {
196 break;
197 }
198
199 busyFlag = getBusyFlag(lidarliteAddress);
200
201 // Increment busyCounter for timeout
202 busyCounter++;
203 }
204
205 // bailout reports error over serial
206 if (busyCounter > 9999)
207 {
208 Serial.println("> bailing out of waitForBusy()");
209 }
210} /* LIDARLite_v3HP::waitForBusy */

222uint8_t LIDARLite_v3HP::getBusyFlag(uint8_t lidarliteAddress)
223{
224 uint8_t busyFlag; // busyFlag monitors when the device is done with a measurement
225
226 // Read status register to check busy flag
227 read(0x01, &busyFlag, 1, lidarliteAddress);
228
229 // STATUS bit 0 is busyFlag
230 busyFlag &= 0x01;

235/*------------------------------------------------------------------------------
236 Read Distance
237
238 Read and return result of distance measurement.
239
240 Process
241 ------------------------------------------------------------------------------
242 1. Read two bytes from register 0x8f and save
243 2. Shift the first value from 0x8f << 8 and add to second value from 0x8f.
244 The result is the measured distance in centimeters.
245
246 Parameters
247 ------------------------------------------------------------------------------
248 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
249 operating manual for instructions.
250------------------------------------------------------------------------------*/
251uint16_t LIDARLite_v3HP::readDistance(uint8_t lidarliteAddress)
252{
253 uint16_t distance;
254 uint8_t dataBytes[2];
255
256 // Read two bytes from register 0x0f and 0x10 (autoincrement)
257 read(0x0f, dataBytes, 2, lidarliteAddress);
258
259 // Shift high byte and add to low byte
260 distance = (dataBytes[0] << 8) | dataBytes[1];
261
262 return (distance);
263} /* LIDARLite_v3HP::readDistance */

265/*------------------------------------------------------------------------------
266 Reset Reference Filter
267
268 In some scenarios, power-on transients in the LIDAR-Lite v3HP can result in
269 initial measurements that may be a few centimeters short of actual distance.
270 This symptom will eventually rectify itself after a few hundred measurements.
271 This symptom can also be rectified more quickly by resetting the unit's internal
272 reference filter. The process here illustrates how to disable the internal
273 reference filter, trigger a measurement (forcing re-init of the reference
274 filter), and then re-enable the filter.
275
276 Process
277 ------------------------------------------------------------------------------
278 1. Disable the LIDAR-Lite reference filter
279 2. Set reference integration count to max
280 3. Trigger a measurement
281 4. Restore reference integration count
282 5. Re-enable reference filter
283
284 Parameters
285 ------------------------------------------------------------------------------
286 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
287 operating manual for instructions.
288------------------------------------------------------------------------------*/
289void LIDARLite_v3HP::resetReferenceFilter(uint8_t lidarliteAddress)
290{
291 uint8_t dataBytes[2];
292 uint8_t acqConfigReg;
293 uint8_t refCountMax;
294
295 // Set bit 4 of the acquisition configuration register (disable reference filter)
296 read(0x04, dataBytes, 1, lidarliteAddress); // Read address 0x04 (acquisition config register)
297 acqConfigReg = dataBytes[0]; // store for later restoration
298 dataBytes[0] = dataBytes[0] | 0x10; // turn on disable of ref filter
299 write(0x04, dataBytes, 1, lidarliteAddress); // write it back
300
301 // Set reference integration count to max
302 read(0x12, dataBytes, 1, lidarliteAddress); // Read address 0x12 (ref integration count)
303 refCountMax = dataBytes[0]; // store for later restoration
304 dataBytes[0] = 0xff; // we want to reference to overflow quickly
305 write(0x12, dataBytes, 1, lidarliteAddress); // write ref integration count
306
307 // Trigger a measurement
308 waitForBusy(lidarliteAddress);
309 takeRange(lidarliteAddress);
310 waitForBusy(lidarliteAddress);
311 // ... no need to read the distance, it is immaterial
312
313 // Restore previous reference integration count
314 dataBytes[0] = refCountMax;
315 write(0x12, dataBytes, 1, lidarliteAddress);
316
317 // Restore previous acquisition configuration register (re-enabling reference filter)
318 dataBytes[0] = acqConfigReg;
319 write(0x04, dataBytes, 1, lidarliteAddress);
320} /* LIDARLite_v3HP::resetReferenceFilter */

322/*------------------------------------------------------------------------------
323 Write
324
325 Perform I2C write to device. The I2C peripheral in the LidarLite v3 HP
326 will receive multiple bytes in one I2C transmission. The first byte is
327 always the register address. The the bytes that follow will be written
328 into the specified register address first and then the internal address
329 in the Lidar Lite will be auto-incremented for all following bytes.
330
331 Parameters
332 ------------------------------------------------------------------------------
333 regAddr: register address to write to
334 dataBytes: pointer to array of bytes to write
335 numBytes: number of bytes in 'dataBytes' array to write
336 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
337 operating manual for instructions.
338------------------------------------------------------------------------------*/
339void LIDARLite_v3HP::write(uint8_t regAddr, uint8_t * dataBytes,
340 uint16_t numBytes, uint8_t lidarliteAddress)
341{
342 int nackCatcher;
343
344 Wire.beginTransmission((int) lidarliteAddress);
345
346 // Wire.write Syntax
347 // -----------------------------------------------------------------
348 // Wire.write(value) - a value to send as a single byte
349 // Wire.write(string) - a string to send as a series of bytes
350 // Wire.write(data, length) - an array of data to send as bytes
351
352 // First byte of every write sets the LidarLite's internal register address pointer
353 Wire.write((int) regAddr);
354
355 // Subsequent bytes are data writes
356 Wire.write(dataBytes, (int) numBytes);
357
358 // A nack means the device is not responding. Report the error over serial.
359 nackCatcher = Wire.endTransmission();
360 if (nackCatcher != 0)
361 {
362 Serial.println("> nack");
363 }
364
365 delayMicroseconds(100); // 100 us delay for robustness with successive reads and writes
366} /* LIDARLite_v3HP::write */

368/*------------------------------------------------------------------------------
369 Read
370
371 Perform I2C read from device. The I2C peripheral in the LidarLite v3 HP
372 will send multiple bytes in one I2C transmission. The register address must
373 be set up by a previous I2C write. The bytes that follow will be read
374 from the specified register address first and then the internal address
375 pointer in the Lidar Lite will be auto-incremented for following bytes.
376
377 Will detect an unresponsive device and report the error over serial.
378
379 Parameters
380 ------------------------------------------------------------------------------
381 regAddr: register address to write to
382 dataBytes: pointer to array of bytes to write
383 numBytes: number of bytes in 'dataBytes' array to write
384 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
385 operating manual for instructions.
386------------------------------------------------------------------------------*/
387void LIDARLite_v3HP::read(uint8_t regAddr, uint8_t * dataBytes,
388 uint16_t numBytes, uint8_t lidarliteAddress)
389{
390 uint16_t i = 0;
391 int nackCatcher = 0;
392
393 // Set the internal register address pointer in the Lidar Lite
394 Wire.beginTransmission((int) lidarliteAddress);
395 Wire.write((int) regAddr); // Set the register to be read
396
397 // A nack means the device is not responding, report the error over serial
398 nackCatcher = Wire.endTransmission(false); // false means perform repeated start
399 if (nackCatcher != 0)
400 {
401 Serial.println("> nack");
402 }
403
404 // Perform read, save in dataBytes array
405 Wire.requestFrom((int)lidarliteAddress, (int) numBytes);
406 if ((int) numBytes <= Wire.available())
407 {
408 while (i < numBytes)
409 {
410 dataBytes[i] = (uint8_t) Wire.read();
411 i++;
412 }
413 }
414
415} /* LIDARLite_v3HP::read */

417/*------------------------------------------------------------------------------
418 Correlation Record To Serial
419
420 The correlation record used to calculate distance can be read from the device.
421 It has a bipolar wave shape, transitioning from a positive going portion to a
422 roughly symmetrical negative going pulse. The point where the signal crosses
423 zero represents the effective delay for the reference and return signals.
424
425 Process
426 ------------------------------------------------------------------------------
427 1. Take a distance reading (there is no correlation record without at least
428 one distance reading being taken)
429 2. Set test mode select by writing 0x07 to register 0x40
430 3. For as many readings as you want to take (max is 1024)
431 1. Read two bytes from 0x52
432 2. The Low byte is the value from the record
433 3. The high byte is the sign from the record
434
435 Parameters
436 ------------------------------------------------------------------------------
437 numberOfReadings: Default: 1024. Maximum of 1024
438 lidarliteAddress: Default 0x62. Fill in new address here if changed. See
439 operating manual for instructions.
440------------------------------------------------------------------------------*/
441void LIDARLite_v3HP::correlationRecordToSerial(
442 uint16_t numberOfReadings, uint8_t lidarliteAddress)
443{
444 uint16_t i = 0;
445 uint8_t dataBytes[2]; // Array to store read / write data
446 int16_t correlationValue = 0; // Var to store value of correlation record
447
448 // Test mode enable
449 dataBytes[0] = 0x07;
450 write(0x40, dataBytes, 1, lidarliteAddress);
451
452 for (i=0 ; i<numberOfReadings ; i++)
453 {
454 read(0x52, dataBytes, 2, lidarliteAddress);
455
456 // Low byte is the value of the correlation record
457 correlationValue = (uint16_t) dataBytes[0];
458
459 // if upper byte lsb is one, the value is negative
460 // so here we test to artifically sign extend the data
461 if ( (int) dataBytes[1] == 1)
462 {
463 correlationValue |= 0xff00;
464 }
465 Serial.println(correlationValue);
466 }
467
468 // test mode disable
469 dataBytes[0] = 0;
470 write(0x40, dataBytes, 1, lidarliteAddress);
471} /* LIDARLite_v3HP::correlationRecordToSerial */

473/*------------------------------------------------------------------------------
474 Correlation Peak Stack Read
475
476 Following a distance measurement, the correlation record is searched and a peak
477 stack table is built. The nine total entries in the peak stack table are
478 readable via registers, as illustrated below. This function reads all stack
479 entries, calculates distances from those entries, and then returns arrays
480 containing the peak magnitudes and associated distances.
481
482 - The first stack entry represents internal reference measurement data.
483 - The next eight entries represent the largest peaks in the correlation record.
484 - The eight range entries are sorted largest peak to smallest peak.
485 - The first of these eight entries is assumed to be the intended target
486 and is used to calculate the distance data stored in I2C registers.
487 - The remaining entries most often represent system noise.
488
489 Each stack entry is comprised of three 16-bit values (6 total bytes)
490 1st value = Peak Strength (Magnitude of Peak)
491 2nd value = Zero crossing address (Coarse Distance)
492 3rd value = Interpolated value (Fine Distance)
493
494 Note: There are calibration and compensation circuits not available
495 outside the LIDAR-Lite system, but the following shows how to
496 quickly extract multiple signatures from the correlation results.
497 Using this method does not result in data that matches the
498 device distance registers exactly, but it is representative.
499
500 Note: Using this method, some negative distances can be produced
501 if noise artifacts exist very early in the correlation record.
502 This is normal. Any negative distances should be ignored.
503
504 Note: LIDAR-Lite must be idle (not BUSY) in order to perform this function.
505
506 Process
507 ------------------------------------------------------------------------------
508 1. Take a distance reading (there is no peak stack data available before
509 at least one distance reading is performed)
510 2. Read stack entries
511
512 Parameters
513 ------------------------------------------------------------------------------
514 peakArray: Pointer to an array of eight 16-bit unsigned data values
515 for storing the Peak Strengths (Magnitude) from the stack
516 distArray: Pointer to an array of eight 16-bit unsigned data values
517 for storing the effective distance associated with each peak
518 lidarliteAddress: Default 0x62. Fill in the new address here if changed. See
519 operating manual for instructions.
520------------------------------------------------------------------------------*/
521void LIDARLite_v3HP::peakStackRead(
522 int16_t * peakArray, int16_t * distArray, uint8_t lidarliteAddress)
523{
524 #define RECORD_OFFSET ((63*18) + 1)
525 #define FREQ_ADJUST (1.067)
526
527 uint8_t idx;
528 uint8_t dataBytes[2];
529 int16_t peakVal;
530 int16_t coarseDist; // Zero crossing address (Coarse Distance)
531 int16_t fineDist; // Interpolated value (Fine Distance)
532 int16_t LLref;
533 int16_t LLtarget;
534
535 // Reset the peak stack internal address pointer
536 dataBytes[0] = 1;
537 write(0x26, dataBytes, 1, lidarliteAddress);
538
539 // Each time through the loop read one peak stack "entry."
540 // First time through the loop always retrieves reference data entry.
541 for (idx = 0 ; idx < 9 ; idx++)
542 {
543 // *** Read "Peak Strength"
544 read(0x26, dataBytes, 2, lidarliteAddress);
545 peakVal = (int16_t) ((dataBytes[0] << 8) + dataBytes[1]);
546
547 // *** Read "Zero Crossing Address"
548 read(0x26, dataBytes, 2, lidarliteAddress);
549 coarseDist = (int16_t) ((dataBytes[0] << 8) + dataBytes[1]);
550
551 // *** Read "Interpolated Value"
552 read(0x26, dataBytes, 2, lidarliteAddress);
553 fineDist = (int16_t) ((dataBytes[0] << 8) + dataBytes[1]);
554
555 // ---------- Calculate distance from peak stack data -------------
556
557 LLtarget = (coarseDist * 18) + fineDist;
558
559 // The first entry in the peak stack contains reference data
560 if (idx == 0)
561 LLref = LLtarget;
562
563 // The remaining eight entries contain data corresponding to the
564 // eight largest peaks in the correlation record
565 if (idx != 0)
566 {
567 peakArray[idx-1] = peakVal;
568 distArray[idx-1] = ((LLtarget - RECORD_OFFSET) - LLref) * FREQ_ADJUST;
569 }
570 }
571} /* LIDARLite_v3HP::peakStackRead */
