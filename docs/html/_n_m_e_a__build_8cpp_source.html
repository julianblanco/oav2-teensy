591void Adafruit_GPS::addChecksum(char *buff) {
592 char cs = 0;
593 int i = 1;
594 while (buff[i]) {
595 cs ^= buff[i];
596 i++;
597 }
598 sprintf(buff, "%s*%02X", buff, cs);
599}
