32bool PX4Flow::update()
33{
34 //send 0x0 to PX4FLOW module and receive back 22 Bytes data
35 Wire.beginTransmission(PX4FLOW_ADDRESS);
36 Wire.write(0x0);
37 Wire.endTransmission();
38
39 // request 22 bytes from the module
40 Wire.requestFrom(PX4FLOW_ADDRESS, 22);
41
42 // wait for all data to be available
43 if (!wait(22)) {
44 return false;
45 }
46
47 // read the data
48 frame.frame_count = read16();
49 frame.pixel_flow_x_sum = read16();
50 frame.pixel_flow_y_sum = read16();
51 frame.flow_comp_m_x = read16();
52 frame.flow_comp_m_y = read16();
53 frame.qual = read16();
54 frame.gyro_x_rate = read16();
55 frame.gyro_y_rate = read16();
56 frame.gyro_z_rate = read16();
57 frame.gyro_range = read8();
58 frame.sonar_timestamp = read8();
59 frame.ground_distance = read16();
60
61 // if too many bytes are available, we drain in order to be synched
62 // on next read
63 if(Wire.available()) {
64 #if PX4FLOW_DEBUG == true
65 {
66 Serial.println("ERROR [PX4Flow] : Too many bytes available.");
67 }
68 #endif
69 while(Wire.available()) {Wire.read();}
70 }
71
72 return true;
73}

75bool PX4Flow::update_integral()
76{
77 //send 0x16 to PX4FLOW module and receive back 25 Bytes data
78 Wire.beginTransmission(PX4FLOW_ADDRESS);
79 Wire.write(0x16);
80 Wire.endTransmission();
81
82 // request 25 bytes from the module
83 Wire.requestFrom(PX4FLOW_ADDRESS, 26);
84
85 // wait for all data to be available
86 // TODO we could manage a timeout in order not to block
87 // the loop when no component is connected
88 if (!wait(26)) {
89 return false;
90 }
91
92 // read the data
93 iframe.frame_count_since_last_readout = read16();
94 iframe.pixel_flow_x_integral = read16();
95 iframe.pixel_flow_y_integral = read16();
96 iframe.gyro_x_rate_integral = read16();
97 iframe.gyro_y_rate_integral = read16();
98 iframe.gyro_z_rate_integral = read16();
99 iframe.integration_timespan = read32();
100 iframe.sonar_timestamp = read32();
101 iframe.ground_distance = read16();
102 iframe.gyro_temperature = read16();
103 iframe.quality = read8();
104
105 // This is due to the lack of structure packing
106 // in the PX4Flow code.
107 read8();
108
109 // if too many bytes are available, we drain in order to be synched
110 // on next read
111 if(Wire.available()) {
112 #if PX4FLOW_DEBUG == true
113 {
114 Serial.println("ERROR [PX4Flow] : Too many bytes available.");
115 }
116 #endif
117 while(Wire.available()) {Wire.read();}
118 }
119
120 return true;
121}

226bool PX4Flow::wait(int count) {
227 unsigned long now = millis();
228 while(Wire.available() < count) {
229 if ((millis() - now) > PX4FLOW_TIMEOUT) {
230 #if PX4FLOW_DEBUG == true
231 {
232 Serial.println("ERROR [PX4Flow] : Timeout reading PX4_Flow.");
233 }
234 #endif
235 return false;
236 }
237 delay(1);
238 }
239 return true;
240}
