178uint16_t BNO080::getReadings(void)
179{
180 //If we have an interrupt pin connection available, check if data is available.
181 //If int pin is not set, then we'll rely on receivePacket() to timeout
182 //See issue 13: https://github.com/sparkfun/SparkFun_BNO080_Arduino_Library/issues/13
183 if (_int != 255)
184 {
185 if (digitalRead(_int) == HIGH)
186 return 0;
187 }
188
189 if (receivePacket() == true)
190 {
191 //Check to see if this packet is a sensor reporting its data to us
192 if (shtpHeader[2] == CHANNEL_REPORTS && shtpData[0] == SHTP_REPORT_BASE_TIMESTAMP)
193 {
194 return parseInputReport(); //This will update the rawAccelX, etc variables depending on which feature report is found
195 }
196 else if (shtpHeader[2] == CHANNEL_CONTROL)
197 {
198 return parseCommandReport(); //This will update responses to commands, calibrationStatus, etc.
199 }
200 else if(shtpHeader[2] == CHANNEL_GYRO)
201 {
202 return parseInputReport(); //This will update the rawAccelX, etc variables depending on which feature report is found
203 }
204 }
205 return 0;

249//This function pulls the data from the input report
250//The input reports vary in length so this function stores the various 16-bit values as globals
251
252//Unit responds with packet that contains the following:
253//shtpHeader[0:3]: First, a 4 byte header
254//shtpData[0:4]: Then a 5 byte timestamp of microsecond clicks since reading was taken
255//shtpData[5 + 0]: Then a feature report ID (0x01 for Accel, 0x05 for Rotation Vector)
256//shtpData[5 + 1]: Sequence number (See 6.5.18.2)
257//shtpData[5 + 2]: Status
258//shtpData[3]: Delay
259//shtpData[4:5]: i/accel x/gyro x/etc
260//shtpData[6:7]: j/accel y/gyro y/etc
261//shtpData[8:9]: k/accel z/gyro z/etc
262//shtpData[10:11]: real/gyro temp/etc
263//shtpData[12:13]: Accuracy estimate
264uint16_t BNO080::parseInputReport(void)
265{
266 //Calculate the number of data bytes in this packet
267 int16_t dataLength = ((uint16_t)shtpHeader[1] << 8 | shtpHeader[0]);
268 dataLength &= ~(1 << 15); //Clear the MSbit. This bit indicates if this package is a continuation of the last.
269 //Ignore it for now. TODO catch this as an error and exit
270
271 dataLength -= 4; //Remove the header bytes from the data count
272
273 timeStamp = ((uint32_t)shtpData[4] << (8 * 3)) | ((uint32_t)shtpData[3] << (8 * 2)) | ((uint32_t)shtpData[2] << (8 * 1)) | ((uint32_t)shtpData[1] << (8 * 0));
274
275 // The gyro-integrated input reports are sent via the special gyro channel and do no include the usual ID, sequence, and status fields
276 if(shtpHeader[2] == CHANNEL_GYRO) {
277 rawQuatI = (uint16_t)shtpData[1] << 8 | shtpData[0];
278 rawQuatJ = (uint16_t)shtpData[3] << 8 | shtpData[2];
279 rawQuatK = (uint16_t)shtpData[5] << 8 | shtpData[4];
280 rawQuatReal = (uint16_t)shtpData[7] << 8 | shtpData[6];
281 rawFastGyroX = (uint16_t)shtpData[9] << 8 | shtpData[8];
282 rawFastGyroY = (uint16_t)shtpData[11] << 8 | shtpData[10];
283 rawFastGyroZ = (uint16_t)shtpData[13] << 8 | shtpData[12];
284
285 return SENSOR_REPORTID_GYRO_INTEGRATED_ROTATION_VECTOR;
286 }
287
288 uint8_t status = shtpData[5 + 2] & 0x03; //Get status bits
289 uint16_t data1 = (uint16_t)shtpData[5 + 5] << 8 | shtpData[5 + 4];
290 uint16_t data2 = (uint16_t)shtpData[5 + 7] << 8 | shtpData[5 + 6];
291 uint16_t data3 = (uint16_t)shtpData[5 + 9] << 8 | shtpData[5 + 8];
292 uint16_t data4 = 0;
293 uint16_t data5 = 0; //We would need to change this to uin32_t to capture time stamp value on Raw Accel/Gyro/Mag reports
294
295 if (dataLength - 5 > 9)
296 {
297 data4 = (uint16_t)shtpData[5 + 11] << 8 | shtpData[5 + 10];
298 }
299 if (dataLength - 5 > 11)
300 {
301 data5 = (uint16_t)shtpData[5 + 13] << 8 | shtpData[5 + 12];
302 }
303
304 //Store these generic values to their proper global variable
305 if (shtpData[5] == SENSOR_REPORTID_ACCELEROMETER)
306 {
307 accelAccuracy = status;
308 rawAccelX = data1;
309 rawAccelY = data2;
310 rawAccelZ = data3;
311 }
312 else if (shtpData[5] == SENSOR_REPORTID_LINEAR_ACCELERATION)
313 {
314 accelLinAccuracy = status;
315 rawLinAccelX = data1;
316 rawLinAccelY = data2;
317 rawLinAccelZ = data3;
318 }
319 else if (shtpData[5] == SENSOR_REPORTID_GYROSCOPE)
320 {
321 gyroAccuracy = status;
322 rawGyroX = data1;
323 rawGyroY = data2;
324 rawGyroZ = data3;
325 }
326 else if (shtpData[5] == SENSOR_REPORTID_MAGNETIC_FIELD)
327 {
328 magAccuracy = status;
329 rawMagX = data1;
330 rawMagY = data2;
331 rawMagZ = data3;
332 }
333 else if (shtpData[5] == SENSOR_REPORTID_ROTATION_VECTOR ||
334 shtpData[5] == SENSOR_REPORTID_GAME_ROTATION_VECTOR ||
335 shtpData[5] == SENSOR_REPORTID_AR_VR_STABILIZED_ROTATION_VECTOR ||
336 shtpData[5] == SENSOR_REPORTID_AR_VR_STABILIZED_GAME_ROTATION_VECTOR)
337 {
338 quatAccuracy = status;
339 rawQuatI = data1;
340 rawQuatJ = data2;
341 rawQuatK = data3;
342 rawQuatReal = data4;
343
344 //Only available on rotation vector and ar/vr stabilized rotation vector,
345 // not game rot vector and not ar/vr stabilized rotation vector
346 rawQuatRadianAccuracy = data5;
347 }
348 else if (shtpData[5] == SENSOR_REPORTID_TAP_DETECTOR)
349 {
350 tapDetector = shtpData[5 + 4]; //Byte 4 only
351 }
352 else if (shtpData[5] == SENSOR_REPORTID_STEP_COUNTER)
353 {
354 stepCount = data3; //Bytes 8/9
355 }
356 else if (shtpData[5] == SENSOR_REPORTID_STABILITY_CLASSIFIER)
357 {
358 stabilityClassifier = shtpData[5 + 4]; //Byte 4 only
359 }
360 else if (shtpData[5] == SENSOR_REPORTID_PERSONAL_ACTIVITY_CLASSIFIER)
361 {
362 activityClassifier = shtpData[5 + 5]; //Most likely state
363
364 //Load activity classification confidences into the array
365 for (uint8_t x = 0; x < 9; x++) //Hardcoded to max of 9. TODO - bring in array size
366 _activityConfidences[x] = shtpData[5 + 6 + x]; //5 bytes of timestamp, byte 6 is first confidence byte
367 }
368 else if (shtpData[5] == SENSOR_REPORTID_RAW_ACCELEROMETER)
369 {
370 memsRawAccelX = data1;
371 memsRawAccelY = data2;
372 memsRawAccelZ = data3;
373 }
374 else if (shtpData[5] == SENSOR_REPORTID_RAW_GYROSCOPE)
375 {
376 memsRawGyroX = data1;
377 memsRawGyroY = data2;
378 memsRawGyroZ = data3;
379 }
380 else if (shtpData[5] == SENSOR_REPORTID_RAW_MAGNETOMETER)
381 {
382 memsRawMagX = data1;
383 memsRawMagY = data2;
384 memsRawMagZ = data3;
385 }
386 else if (shtpData[5] == SHTP_REPORT_COMMAND_RESPONSE)
387 {
388 if (_printDebug == true)
389 {
390 _debugPort->println(F("!"));
391 }
392 //The BNO080 responds with this report to command requests. It's up to use to remember which command we issued.
393 uint8_t command = shtpData[5 + 2]; //This is the Command byte of the response
394
395 if (command == COMMAND_ME_CALIBRATE)
396 {
397 if (_printDebug == true)
398 {
399 _debugPort->println(F("ME Cal report found!"));
400 }
401 calibrationStatus = shtpData[5 + 5]; //R0 - Status (0 = success, non-zero = fail)
402 }
403 }
404 else
405 {
406 //This sensor report ID is unhandled.
407 //See reference manual to add additional feature reports as needed
408 return 0;
409 }
410
411 //TODO additional feature reports may be strung together. Parse them all.
412 return shtpData[5];
413}

415// Quaternion to Euler conversion
416// https://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles
417// https://github.com/sparkfun/SparkFun_MPU-9250-DMP_Arduino_Library/issues/5#issuecomment-306509440
418// Return the roll (rotation around the x-axis) in Radians
419float BNO080::getRoll()
420{
421 float dqw = getQuatReal();
422 float dqx = getQuatI();
423 float dqy = getQuatJ();
424 float dqz = getQuatK();
425
426 float norm = sqrt(dqw*dqw + dqx*dqx + dqy*dqy + dqz*dqz);
427 dqw = dqw/norm;
428 dqx = dqx/norm;
429 dqy = dqy/norm;
430 dqz = dqz/norm;
431
432 float ysqr = dqy * dqy;
433
434 // roll (x-axis rotation)
435 float t0 = +2.0 * (dqw * dqx + dqy * dqz);
436 float t1 = +1.0 - 2.0 * (dqx * dqx + ysqr);
437 float roll = atan2(t0, t1);
438
439 return (roll);
440}

442// Return the pitch (rotation around the y-axis) in Radians
443float BNO080::getPitch()
444{
445 float dqw = getQuatReal();
446 float dqx = getQuatI();
447 float dqy = getQuatJ();
448 float dqz = getQuatK();
449
450 float norm = sqrt(dqw*dqw + dqx*dqx + dqy*dqy + dqz*dqz);
451 dqw = dqw/norm;
452 dqx = dqx/norm;
453 dqy = dqy/norm;
454 dqz = dqz/norm;
455
456 float ysqr = dqy * dqy;
457
458 // pitch (y-axis rotation)
459 float t2 = +2.0 * (dqw * dqy - dqz * dqx);
460 t2 = t2 > 1.0 ? 1.0 : t2;
461 t2 = t2 < -1.0 ? -1.0 : t2;
462 float pitch = asin(t2);
463
464 return (pitch);
465}

467// Return the yaw / heading (rotation around the z-axis) in Radians
468float BNO080::getYaw()
469{
470 float dqw = getQuatReal();
471 float dqx = getQuatI();
472 float dqy = getQuatJ();
473 float dqz = getQuatK();
474
475 float norm = sqrt(dqw*dqw + dqx*dqx + dqy*dqy + dqz*dqz);
476 dqw = dqw/norm;
477 dqx = dqx/norm;
478 dqy = dqy/norm;
479 dqz = dqz/norm;
480
481 float ysqr = dqy * dqy;
482
483 // yaw (z-axis rotation)
484 float t3 = +2.0 * (dqw * dqz + dqx * dqy);
485 float t4 = +1.0 - 2.0 * (ysqr + dqz * dqz);
486 float yaw = atan2(t3, t4);
487
488 return (yaw);
489}

904//Ask the sensor for data from the Flash Record System
905//See 6.3.6 page 40, FRS Read Request
906void BNO080::frsReadRequest(uint16_t recordID, uint16_t readOffset, uint16_t blockSize)
907{
908 shtpData[0] = SHTP_REPORT_FRS_READ_REQUEST; //FRS Read Request
909 shtpData[1] = 0; //Reserved
910 shtpData[2] = (readOffset >> 0) & 0xFF; //Read Offset LSB
911 shtpData[3] = (readOffset >> 8) & 0xFF; //Read Offset MSB
912 shtpData[4] = (recordID >> 0) & 0xFF; //FRS Type LSB
913 shtpData[5] = (recordID >> 8) & 0xFF; //FRS Type MSB
914 shtpData[6] = (blockSize >> 0) & 0xFF; //Block size LSB
915 shtpData[7] = (blockSize >> 8) & 0xFF; //Block size MSB
916
917 //Transmit packet on channel 2, 8 bytes
918 sendPacket(CHANNEL_CONTROL, 8);
919}

921//Given a sensor or record ID, and a given start/stop bytes, read the data from the Flash Record System (FRS) for this sensor
922//Returns true if metaData array is loaded successfully
923//Returns false if failure
924bool BNO080::readFRSdata(uint16_t recordID, uint8_t startLocation, uint8_t wordsToRead)
925{
926 uint8_t spot = 0;
927
928 //First we send a Flash Record System (FRS) request
929 frsReadRequest(recordID, startLocation, wordsToRead); //From startLocation of record, read a # of words
930
931 //Read bytes until FRS reports that the read is complete
932 while (1)
933 {
934 //Now we wait for response
935 while (1)
936 {
937 uint8_t counter = 0;
938 while (receivePacket() == false)
939 {
940 if (counter++ > 100)
941 return (false); //Give up
942 delay(1);
943 }
944
945 //We have the packet, inspect it for the right contents
946 //See page 40. Report ID should be 0xF3 and the FRS types should match the thing we requested
947 if (shtpData[0] == SHTP_REPORT_FRS_READ_RESPONSE)
948 if (((((uint16_t)shtpData[13]) << 8) | shtpData[12]) == recordID)
949 break; //This packet is one we are looking for
950 }
951
952 uint8_t dataLength = shtpData[1] >> 4;
953 uint8_t frsStatus = shtpData[1] & 0x0F;
954
955 uint32_t data0 = (uint32_t)shtpData[7] << 24 | (uint32_t)shtpData[6] << 16 | (uint32_t)shtpData[5] << 8 | (uint32_t)shtpData[4];
956 uint32_t data1 = (uint32_t)shtpData[11] << 24 | (uint32_t)shtpData[10] << 16 | (uint32_t)shtpData[9] << 8 | (uint32_t)shtpData[8];
957
958 //Record these words to the metaData array
959 if (dataLength > 0)
960 {
961 metaData[spot++] = data0;
962 }
963 if (dataLength > 1)
964 {
965 metaData[spot++] = data1;
966 }
967
968 if (spot >= MAX_METADATA_SIZE)
969 {
970 if (_printDebug == true)
971 _debugPort->println(F("metaData array over run. Returning."));
972 return (true); //We have run out of space in our array. Bail.
973 }
974
975 if (frsStatus == 3 || frsStatus == 6 || frsStatus == 7)
976 {
977 return (true); //FRS status is read completed! We're done!
978 }
979 }
980}

1261void BNO080::sendCommand(uint8_t command)
1262{

1264 shtpData[1] = commandSequenceNumber++; //Increments automatically each function call
1265 shtpData[2] = command; //Command
