65 void fromAxisAngle(const Vector<3> &axis, double theta) {
66 _w = cos(theta / 2);
67 // only need to calculate sine of half theta once
68 double sht = sin(theta / 2);
69 _x = axis.x() * sht;
70 _y = axis.y() * sht;
71 _z = axis.z() * sht;
72 }

74 void fromMatrix(const Matrix<3> &m) {
75 double tr = m.trace();
76
77 double S;
78 if (tr > 0) {
79 S = sqrt(tr + 1.0) * 2;
80 _w = 0.25 * S;
81 _x = (m(2, 1) - m(1, 2)) / S;
82 _y = (m(0, 2) - m(2, 0)) / S;
83 _z = (m(1, 0) - m(0, 1)) / S;
84 } else if (m(0, 0) > m(1, 1) && m(0, 0) > m(2, 2)) {
85 S = sqrt(1.0 + m(0, 0) - m(1, 1) - m(2, 2)) * 2;
86 _w = (m(2, 1) - m(1, 2)) / S;
87 _x = 0.25 * S;
88 _y = (m(0, 1) + m(1, 0)) / S;
89 _z = (m(0, 2) + m(2, 0)) / S;
90 } else if (m(1, 1) > m(2, 2)) {
91 S = sqrt(1.0 + m(1, 1) - m(0, 0) - m(2, 2)) * 2;
92 _w = (m(0, 2) - m(2, 0)) / S;
93 _x = (m(0, 1) + m(1, 0)) / S;
94 _y = 0.25 * S;
95 _z = (m(1, 2) + m(2, 1)) / S;
96 } else {
97 S = sqrt(1.0 + m(2, 2) - m(0, 0) - m(1, 1)) * 2;
98 _w = (m(1, 0) - m(0, 1)) / S;
99 _x = (m(0, 2) + m(2, 0)) / S;
100 _y = (m(1, 2) + m(2, 1)) / S;
101 _z = 0.25 * S;
102 }
103 }

105 void toAxisAngle(Vector<3> &axis, double &angle) const {
106 double sqw = sqrt(1 - _w * _w);
107 if (sqw == 0) // it's a singularity and divide by zero, avoid
108 return;
109
110 angle = 2 * acos(_w);
111 axis.x() = _x / sqw;
112 axis.y() = _y / sqw;
113 axis.z() = _z / sqw;
114 }

116 Matrix<3> toMatrix() const {
117 Matrix<3> ret;
118 ret.cell(0, 0) = 1 - 2 * _y * _y - 2 * _z * _z;
119 ret.cell(0, 1) = 2 * _x * _y - 2 * _w * _z;
120 ret.cell(0, 2) = 2 * _x * _z + 2 * _w * _y;
121
122 ret.cell(1, 0) = 2 * _x * _y + 2 * _w * _z;
123 ret.cell(1, 1) = 1 - 2 * _x * _x - 2 * _z * _z;
124 ret.cell(1, 2) = 2 * _y * _z - 2 * _w * _x;
125
126 ret.cell(2, 0) = 2 * _x * _z - 2 * _w * _y;
127 ret.cell(2, 1) = 2 * _y * _z + 2 * _w * _x;
128 ret.cell(2, 2) = 1 - 2 * _x * _x - 2 * _y * _y;
129 return ret;
130 }

132 // Returns euler angles that represent the quaternion. Angles are
133 // returned in rotation order and right-handed about the specified
134 // axes:
135 //
136 // v[0] is applied 1st about z (ie, roll)
137 // v[1] is applied 2nd about y (ie, pitch)
138 // v[2] is applied 3rd about x (ie, yaw)
139 //
140 // Note that this means result.x() is not a rotation about x;
141 // similarly for result.z().
142 //
143 Vector<3> toEuler() const {
144 Vector<3> ret;
145 double sqw = _w * _w;
146 double sqx = _x * _x;
147 double sqy = _y * _y;
148 double sqz = _z * _z;
149
150 ret.x() = atan2(2.0 * (_x * _y + _z * _w), (sqx - sqy - sqz + sqw));
151 ret.y() = asin(-2.0 * (_x * _z - _y * _w) / (sqx + sqy + sqz + sqw));
152 ret.z() = atan2(2.0 * (_y * _z + _x * _w), (-sqx - sqy + sqz + sqw));
153
154 return ret;
155 }

157 Vector<3> toAngularVelocity(double dt) const {
158 Vector<3> ret;
159 Quaternion one(1.0, 0.0, 0.0, 0.0);
160 Quaternion delta = one - *this;
161 Quaternion r = (delta / dt);
162 r = r * 2;
163 r = r * one;
164
165 ret.x() = r.x();
166 ret.y() = r.y();
167 ret.z() = r.z();
168 return ret;
169 }

171 Vector<3> rotateVector(const Vector<2> &v) const {
172 return rotateVector(Vector<3>(v.x(), v.y()));

175 Vector<3> rotateVector(const Vector<3> &v) const {
176 Vector<3> qv(_x, _y, _z);
177 Vector<3> t = qv.cross(v) * 2.0;
178 return v + t * _w + qv.cross(t);
179 }

181 Quaternion operator*(const Quaternion &q) const {
182 return Quaternion(_w * q._w - _x * q._x - _y * q._y - _z * q._z,
183 _w * q._x + _x * q._w + _y * q._z - _z * q._y,
184 _w * q._y - _x * q._z + _y * q._w + _z * q._x,
185 _w * q._z + _x * q._y - _y * q._x + _z * q._w);
186 }
