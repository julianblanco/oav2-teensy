2216/*
2217 * THIS FUNCTION MUST NOT BE USED FROM APPLICATION CODE. IT IS ONLY
2218 * INTENDED FOR USE WHEN IMPLEMENTING A PORT OF THE SCHEDULER AND IS
2219 * AN INTERFACE WHICH IS FOR THE EXCLUSIVE USE OF THE SCHEDULER.
2220 *
2221 * Called from the real time kernel tick (either preemptive or cooperative),
2222 * this increments the tick count and checks if any tasks that are blocked
2223 * for a finite period required removing from a blocked list and placing on
2224 * a ready list. If a non-zero value is returned then a context switch is
2225 * required because either:
2226 * + A task was removed from a blocked list because its timeout had expired,
2227 * or
2228 * + Time slicing is in use and there is a task of equal priority to the
2229 * currently running task.
2230 */
2231BaseType_t xTaskIncrementTick( void ) PRIVILEGED_FUNCTION;
2232
2233/*
2234 * THIS FUNCTION MUST NOT BE USED FROM APPLICATION CODE. IT IS AN
2235 * INTERFACE WHICH IS FOR THE EXCLUSIVE USE OF THE SCHEDULER.
2236 *
2237 * THIS FUNCTION MUST BE CALLED WITH INTERRUPTS DISABLED.
2238 *
2239 * Removes the calling task from the ready list and places it both
2240 * on the list of tasks waiting for a particular event, and the
2241 * list of delayed tasks. The task will be removed from both lists
2242 * and replaced on the ready list should either the event occur (and
2243 * there be no higher priority tasks waiting on the same event) or
2244 * the delay period expires.
2245 *
2246 * The 'unordered' version replaces the event list item value with the
2247 * xItemValue value, and inserts the list item at the end of the list.
2248 *
2249 * The 'ordered' version uses the existing event list item value (which is the
2250 * owning tasks priority) to insert the list item into the event list is task
2251 * priority order.
2252 *
2253 * @param pxEventList The list containing tasks that are blocked waiting
2254 * for the event to occur.
2255 *
2256 * @param xItemValue The item value to use for the event list item when the
2257 * event list is not ordered by task priority.
2258 *
2259 * @param xTicksToWait The maximum amount of time that the task should wait
2260 * for the event to occur. This is specified in kernel ticks,the constant
2261 * portTICK_PERIOD_MS can be used to convert kernel ticks into a real time
2262 * period.
2263 */
2264void vTaskPlaceOnEventList( List_t * const pxEventList, const TickType_t xTicksToWait ) PRIVILEGED_FUNCTION;
2265void vTaskPlaceOnUnorderedEventList( List_t * pxEventList, const TickType_t xItemValue, const TickType_t xTicksToWait ) PRIVILEGED_FUNCTION;

2280/*
2281 * THIS FUNCTION MUST NOT BE USED FROM APPLICATION CODE. IT IS AN
2282 * INTERFACE WHICH IS FOR THE EXCLUSIVE USE OF THE SCHEDULER.
2283 *
2284 * THIS FUNCTION MUST BE CALLED WITH INTERRUPTS DISABLED.
2285 *
2286 * Removes a task from both the specified event list and the list of blocked
2287 * tasks, and places it on a ready queue.
2288 *
2289 * xTaskRemoveFromEventList()/vTaskRemoveFromUnorderedEventList() will be called
2290 * if either an event occurs to unblock a task, or the block timeout period
2291 * expires.
2292 *
2293 * xTaskRemoveFromEventList() is used when the event list is in task priority
2294 * order. It removes the list item from the head of the event list as that will
2295 * have the highest priority owning task of all the tasks on the event list.
2296 * vTaskRemoveFromUnorderedEventList() is used when the event list is not
2297 * ordered and the event list items hold something other than the owning tasks
2298 * priority. In this case the event list item value is updated to the value
2299 * passed in the xItemValue parameter.
2300 *
2301 * @return pdTRUE if the task being removed has a higher priority than the task
2302 * making the call, otherwise pdFALSE.
2303 */
2304BaseType_t xTaskRemoveFromEventList( const List_t * const pxEventList ) PRIVILEGED_FUNCTION;
