88void IMU::requestdatafrompu6050()
89{
90 //DESCRIPTION: Request full dataset from IMU and LP filter gyro and accelerometer data
91 /*
92 * Reads accelerometer and gyro data from IMU as AccX, AccY, AccZ, GyroX, GyroY, GyroZ. These values are scaled
93 * according to the IMU datasheet to put them into correct units of g's and degree/sec. A simple first-order
94 * low-pass filter is used to get rid of high frequency noise in these raw signals. Generally you want to cut
95 * off everything past 80Hz, but if your loop rate is not fast enough, the low pass filter will cause a lag in
96 * the readings. The filter parameters B_gyro and B_accel are set to be good for a 2kHz loop rate. Finally,
97 * the constant errors found in calculate_IMU_error() on startup are subtracted from the readings.
98 */
99 int16_t AcX, AcY, AcZ, GyX, GyY, GyZ;
100 float B_accel = 0.14; //0.01
101 float B_gyro = 0.89; //.9 for 1khz? //0.13 sets cutoff just past 80Hz for about 3000Hz loop rate
102
103 //Get accel data
104 Wire.beginTransmission(MPU);
105 Wire.write(0x3B); //Start with register 0x3B (ACCEL_XOUT_H)
106 Wire.endTransmission(false);
107 Wire.requestFrom(MPU, 6, true); //Read 6 registers total, each axis value is stored in 2 registers
108 //For a range of +-2g, we need to divide the raw values by 16384, according to the datasheet
109 AcX = (Wire.read() << 8 | Wire.read()); //X-axis value
110 AcY = (Wire.read() << 8 | Wire.read()); //Y-axis value
111 AcZ = (Wire.read() << 8 | Wire.read()); //Z-axis value
112 AccX = AcX / 16384.0;
113 AccY = AcY / 16384.0;
114 AccZ = AcZ / 16384.0;
115 //LP filter accelerometer data
116
117 AccX = (1.0 - B_accel) * AccX_prev + B_accel * AccX;
118 AccY = (1.0 - B_accel) * AccY_prev + B_accel * AccY;
119 AccZ = (1.0 - B_accel) * AccZ_prev + B_accel * AccZ;
120 AccX_prev = AccX;
121 AccY_prev = AccY;
122 AccZ_prev = AccZ;
123 //Correct the outputs with the calculated error values
124 AccX = AccX - AccErrorX;
125 AccY = AccY - AccErrorY;
126 AccZ = AccZ - AccErrorZ;
127
128 //Get gyro data
129 Wire.beginTransmission(MPU);
130 Wire.write(0x43); //Gyro data first register address 0x43
131 Wire.endTransmission(false);
132 Wire.requestFrom(MPU, 6, true); //Read 4 registers total, each axis value is stored in 2 registers
133 //For a 250deg/s range we have to divide first the raw value by 131.0, according to the datasheet
134 GyX = (Wire.read() << 8 | Wire.read());
135 GyY = (Wire.read() << 8 | Wire.read());
136 GyZ = (Wire.read() << 8 | Wire.read());
137 GyroX = GyX / 131.0;
138 GyroY = GyY / 131.0;
139 GyroZ = GyZ / 131.0;
140 //LP filter gyro data
141
142 GyroX = (1.0 - B_gyro) * GyroX_prev + B_gyro * GyroX;
143 GyroY = (1.0 - B_gyro) * GyroY_prev + B_gyro * GyroY;
144 GyroZ = (1.0 - B_gyro) * GyroZ_prev + B_gyro * GyroZ;
145 GyroX_prev = GyroX;
146 GyroY_prev = GyroY;
147 GyroZ_prev = GyroZ;
148 //Correct the outputs with the calculated error values
149 GyroX = GyroX - GyroErrorX;
150 GyroY = GyroY - GyroErrorY;
151 GyroZ = GyroZ - GyroErrorZ;
152}

154void IMU::calculate_mpu6050_IMU_error()
155{
156 //DESCRIPTION: Computes IMU error on startup. Note: vehicle should be powered up on flat surface
157 /*
158 * Don't worry too much about what this is doing. The error values it computes are applied to the raw gyro and
159 * accelerometer values AccX, AccY, AccZ, GyroX, GyroY, GyroZ in getIMUdata(). This eliminates drift in the
160 * measurement.
161 */
162 int16_t AcX, AcY, AcZ, GyX, GyY, GyZ;
163
164 //Read accelerometer values 12000 times
165 int c = 0;
166 while (c < 12000)
167 {
168 Wire.beginTransmission(MPU);
169 Wire.write(0x3B);
170 Wire.endTransmission(false);
171 Wire.requestFrom(MPU, 6, true);
172 AcX = (Wire.read() << 8 | Wire.read());
173 AcY = (Wire.read() << 8 | Wire.read());
174 AcZ = (Wire.read() << 8 | Wire.read());
175 AccX = AcX / 16384.0;
176 AccY = AcY / 16384.0;
177 AccZ = AcZ / 16384.0;
178 // Sum all readings
179 AccErrorX = AccErrorX + AccX;
180 AccErrorY = AccErrorY + AccY;
181 AccErrorZ = AccErrorZ + AccZ;
182 c++;
183 }
184 //Divide the sum by 12000 to get the error value
185 AccErrorX = AccErrorX / 12000.0;
186 AccErrorY = AccErrorY / 12000.0;
187 AccErrorZ = AccErrorZ / 12000.0 - 1.0;
188 c = 0;
189 //Read gyro values 12000 times
190 while (c < 12000)
191 {
192 Wire.beginTransmission(MPU);
193 Wire.write(0x43);
194 Wire.endTransmission(false);
195 Wire.requestFrom(MPU, 6, true);
196 GyX = Wire.read() << 8 | Wire.read();
197 GyY = Wire.read() << 8 | Wire.read();
198 GyZ = Wire.read() << 8 | Wire.read();
199 GyroX = GyX / 131.0;
200 GyroY = GyY / 131.0;
201 GyroZ = GyZ / 131.0;
202 // Sum all readings
203 GyroErrorX = GyroErrorX + GyroX;
204 GyroErrorY = GyroErrorY + GyroY;
205 GyroErrorZ = GyroErrorZ + GyroZ;
206 c++;
207 }
208 //Divide the sum by 12000 to get the error value
209 GyroErrorX = GyroErrorX / 12000.0;
210 GyroErrorY = GyroErrorY / 12000.0;
211 GyroErrorZ = GyroErrorZ / 12000.0;
212}

229void IMU::Madgwick(float gx, float gy, float gz, float ax, float ay, float az, float invSampleFreq)
230{
231 //DESCRIPTION: Attitude estimation through sensor fusion
232 /*
233 * This function fuses the accelerometer and gyro readings AccX, AccY, AccZ, GyroX, GyroY, GyroZ for attitude estimation.
234 * Don't worry about the math. There is a tunable parameter called beta in the variable declaration section which basically
235 * adjusts the weight of accelerometer and gyro data in the state estimate. Higher beta leads to noisier estimate, lower
236 * beta leads to slower to respond estimate. It is currently tuned for 2kHz loop rate. This function updates the roll_IMU,
237 * pitch_IMU, and yaw_IMU variables which are in degrees.
238 */
239 float recipNorm;
240 float s0, s1, s2, s3;
241 float qDot1, qDot2, qDot3, qDot4;
242 float _2q0, _2q1, _2q2, _2q3, _4q0, _4q1, _4q2, _8q1, _8q2, q0q0, q1q1, q2q2, q3q3;
243
244 //Convert gyroscope degrees/sec to radians/sec
245 gx *= 0.0174533f;
246 gy *= 0.0174533f;
247 gz *= 0.0174533f;
248
249 //Rate of change of quaternion from gyroscope
250 qDot1 = 0.5f * (-q1 * gx - q2 * gy - q3 * gz);
251 qDot2 = 0.5f * (q0 * gx + q2 * gz - q3 * gy);
252 qDot3 = 0.5f * (q0 * gy - q1 * gz + q3 * gx);
253 qDot4 = 0.5f * (q0 * gz + q1 * gy - q2 * gx);
254
255 //Compute feedback only if accelerometer measurement valid (avoids NaN in accelerometer normalisation)
256 if (!((ax == 0.0f) && (ay == 0.0f) && (az == 0.0f)))
257 {
258 //Normalise accelerometer measurement
259 recipNorm = invSqrt(ax * ax + ay * ay + az * az);
260 ax *= recipNorm;
261 ay *= recipNorm;
262 az *= recipNorm;
263
264 //Auxiliary variables to avoid repeated arithmetic
265 _2q0 = 2.0f * q0;
266 _2q1 = 2.0f * q1;
267 _2q2 = 2.0f * q2;
268 _2q3 = 2.0f * q3;
269 _4q0 = 4.0f * q0;
270 _4q1 = 4.0f * q1;
271 _4q2 = 4.0f * q2;
272 _8q1 = 8.0f * q1;
273 _8q2 = 8.0f * q2;
274 q0q0 = q0 * q0;
275 q1q1 = q1 * q1;
276 q2q2 = q2 * q2;
277 q3q3 = q3 * q3;
278
279 //Gradient decent algorithm corrective step
280 s0 = _4q0 * q2q2 + _2q2 * ax + _4q0 * q1q1 - _2q1 * ay;
281 s1 = _4q1 * q3q3 - _2q3 * ax + 4.0f * q0q0 * q1 - _2q0 * ay - _4q1 + _8q1 * q1q1 + _8q1 * q2q2 + _4q1 * az;
282 s2 = 4.0f * q0q0 * q2 + _2q0 * ax + _4q2 * q3q3 - _2q3 * ay - _4q2 + _8q2 * q1q1 + _8q2 * q2q2 + _4q2 * az;
283 s3 = 4.0f * q1q1 * q3 - _2q1 * ax + 4.0f * q2q2 * q3 - _2q2 * ay;
284 recipNorm = invSqrt(s0 * s0 + s1 * s1 + s2 * s2 + s3 * s3); //normalise step magnitude
285 s0 *= recipNorm;
286 s1 *= recipNorm;
287 s2 *= recipNorm;
288 s3 *= recipNorm;
289
290 //Apply feedback step
291 qDot1 -= beta * s0;
292 qDot2 -= beta * s1;
293 qDot3 -= beta * s2;
294 qDot4 -= beta * s3;
295 }
296
297 //Integrate rate of change of quaternion to yield quaternion
298 q0 += qDot1 * invSampleFreq;
299 q1 += qDot2 * invSampleFreq;
300 q2 += qDot3 * invSampleFreq;
301 q3 += qDot4 * invSampleFreq;
302
303 //Normalise quaternion
304 recipNorm = invSqrt(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
305 q0 *= recipNorm;
306 q1 *= recipNorm;
307 q2 *= recipNorm;
308 q3 *= recipNorm;
309
310 //compute angles
311 roll_IMU = atan2(q0 * q1 + q2 * q3, 0.5f - q1 * q1 - q2 * q2) * 57.29577951; //degrees
312 pitch_IMU = asin(-2.0f * (q1 * q3 - q0 * q2)) * 57.29577951; //degrees
313 yaw_IMU = atan2(q1 * q2 + q0 * q3, 0.5f - q2 * q2 - q3 * q3) * 57.29577951; //degrees
314 // eulerPQR = derive_ang_velocity(const Eigen::Ref<const Eigen::Matrix<double,1,3>>& e_)
315}

317int IMU::calibratempu6050Attitude()
318{
319 //DESCRIPTION: Extra function to calibrate IMU attitude estimate on startup, can be used to warm up everything before entering main loop
320 //Assuming vehicle is powered up on level surface!
321 /*
322 * This function is used on startup to warm up the attitude estimation. Originally used to eliminate additional error in the signal
323 * but no longer used for that purpose as it is not needed on the Teensy. This function is what causes startup to take a few seconds
324 * to boot. The roll_correction and pitch_correction values can be applied to the roll and pitch attitude estimates using
325 * correctRollPitch() in the main loop after the madgwick filter function. Again, we don't use this for that purpose but just to warm
326 * up the IMU and attitude estimation algorithm.
327 */
328 //Warm up IMU and madgwick filter
329 for (int i = 0; i <= 10000; i++)
330 {
331 prev_time = current_time;
332 current_time = micros();
333 dt = (current_time - prev_time) / 1000000.0;
334 getIMUdata();
335 Madgwick(GyroX, GyroY, GyroZ, AccX, AccY, AccZ, dt);
336 loopRate(2000,current_time); //do not exceed 2000Hz
337 }
338 //Grab mean roll and pitch values after everything is warmed up
339 for (int j = 1; j <= 2000; j++)
340 {
341 prev_time = current_time;
342 current_time = micros();
343 dt = (current_time - prev_time) / 1000000.0;
344 getIMUdata();
345 Madgwick(GyroX, GyroY, GyroZ, AccX, AccY, AccZ, dt);
346 roll_correction = roll_IMU + roll_correction;
347 pitch_correction = pitch_IMU + pitch_correction;
348 loopRate(2000,current_time); //do not exceed 2000Hz
349 }
350 //These are applied to roll and pitch after Madgwick filter in main loop if desired using correctRollPitch()
351 roll_correction = roll_correction / 2000.0;
352 pitch_correction = pitch_correction / 2000.0;
353 return (0);
354}
