156void Adafruit_NeoPixel::updateType(neoPixelType t) {
157 bool oldThreeBytesPerPixel = (wOffset == rOffset); // false if RGBW
158
159 wOffset = (t >> 6) & 0b11; // See notes in header file
160 rOffset = (t >> 4) & 0b11; // regarding R/G/B/W offsets
161 gOffset = (t >> 2) & 0b11;
162 bOffset = t & 0b11;
163#if defined(NEO_KHZ400)
164 is800KHz = (t < 256); // 400 KHz flag is 1<<8
165#endif
166
167 // If bytes-per-pixel has changed (and pixel data was previously
168 // allocated), re-allocate to new size. Will clear any data.
169 if(pixels) {
170 bool newThreeBytesPerPixel = (wOffset == rOffset);
171 if(newThreeBytesPerPixel != oldThreeBytesPerPixel) updateLength(numLEDs);
172 }
173}

1341// EasyDMA feature included on the chip. This technique loads the duty
1342// cycle configuration for each cycle when the PWM is enabled. For this
1343// to work we need to store a 16 bit configuration for each bit of the
1344// RGB(W) values in the pixel buffer.

1346// be 100% organic to preserve freshness and high accuracy. Current
1347// parameters are:

1349// * Minimum step time: 62.5ns

1351// * Time for one in high (T1H): 0.75ms

1353// * Frequency: 800Khz
1354// For 400Khz we just double the calculated times.
1355// ---------- BEGIN Constants for the EasyDMA implementation -----------
1356// The PWM starts the duty cycle in LOW. To start with HIGH we
1357// need to set the 15th bit on each register.
1358
1359// WS2812 (rev A) timing is 0.35 and 0.7us
1360//#define MAGIC_T0H 5UL | (0x8000) // 0.3125us
1361//#define MAGIC_T1H 12UL | (0x8000) // 0.75us
1362
1363// WS2812B (rev B) timing is 0.4 and 0.8 us
1364#define MAGIC_T0H 6UL | (0x8000) // 0.375us
1365#define MAGIC_T1H 13UL | (0x8000) // 0.8125us
1366

1368#define MAGIC_T0H_400KHz 8UL | (0x8000) // 0.5us
1369#define MAGIC_T1H_400KHz 19UL | (0x8000) // 1.1875us

1378// implementation is tried.
1379// The nRF52 runs with a fixed clock of 64Mhz. The alternative
1380// implementation is the same as the one used for the Teensy 3.0/1/2 but
1381// with the Nordic SDK HAL & registers syntax.
1382// The number of cycles was hand picked and is guaranteed to be 100%
1383// organic to preserve freshness and high accuracy.
1384// ---------- BEGIN Constants for cycle counter implementation ---------
1385#define CYCLES_800_T0H 18 // ~0.36 uS

2385uint32_t Adafruit_NeoPixel::ColorHSV(uint16_t hue, uint8_t sat, uint8_t val) {
2386
2387 uint8_t r, g, b;
2388
2389 // Remap 0-65535 to 0-1529. Pure red is CENTERED on the 64K rollover;
2390 // 0 is not the start of pure red, but the midpoint...a few values above
2391 // zero and a few below 65536 all yield pure red (similarly, 32768 is the
2392 // midpoint, not start, of pure cyan). The 8-bit RGB hexcone (256 values
2393 // each for red, green, blue) really only allows for 1530 distinct hues
2394 // (not 1536, more on that below), but the full unsigned 16-bit type was
2395 // chosen for hue so that one's code can easily handle a contiguous color
2396 // wheel by allowing hue to roll over in either direction.
2397 hue = (hue * 1530L + 32768) / 65536;
2398 // Because red is centered on the rollover point (the +32768 above,
2399 // essentially a fixed-point +0.5), the above actually yields 0 to 1530,
2400 // where 0 and 1530 would yield the same thing. Rather than apply a
2401 // costly modulo operator, 1530 is handled as a special case below.
2402
2403 // So you'd think that the color "hexcone" (the thing that ramps from
2404 // pure red, to pure yellow, to pure green and so forth back to red,
2405 // yielding six slices), and with each color component having 256
2406 // possible values (0-255), might have 1536 possible items (6*256),
2407 // but in reality there's 1530. This is because the last element in
2408 // each 256-element slice is equal to the first element of the next
2409 // slice, and keeping those in there this would create small
2410 // discontinuities in the color wheel. So the last element of each
2411 // slice is dropped...we regard only elements 0-254, with item 255
2412 // being picked up as element 0 of the next slice. Like this:
2413 // Red to not-quite-pure-yellow is: 255, 0, 0 to 255, 254, 0
2414 // Pure yellow to not-quite-pure-green is: 255, 255, 0 to 1, 255, 0
2415 // Pure green to not-quite-pure-cyan is: 0, 255, 0 to 0, 255, 254
2416 // and so forth. Hence, 1530 distinct hues (0 to 1529), and hence why
2417 // the constants below are not the multiples of 256 you might expect.
2418
2419 // Convert hue to R,G,B (nested ifs faster than divide+mod+switch):
2420 if(hue < 510) { // Red to Green-1
2421 b = 0;
2422 if(hue < 255) { // Red to Yellow-1
2423 r = 255;
2424 g = hue; // g = 0 to 254
2425 } else { // Yellow to Green-1
2426 r = 510 - hue; // r = 255 to 1
2427 g = 255;
2428 }
2429 } else if(hue < 1020) { // Green to Blue-1
2430 r = 0;
2431 if(hue < 765) { // Green to Cyan-1
2432 g = 255;
2433 b = hue - 510; // b = 0 to 254
2434 } else { // Cyan to Blue-1
2435 g = 1020 - hue; // g = 255 to 1
2436 b = 255;
2437 }
2438 } else if(hue < 1530) { // Blue to Red-1
2439 g = 0;
2440 if(hue < 1275) { // Blue to Magenta-1
2441 r = hue - 1020; // r = 0 to 254
2442 b = 255;
2443 } else { // Magenta to Red-1
2444 r = 255;
2445 b = 1530 - hue; // b = 255 to 1
2446 }
2447 } else { // Last 0.5 Red (quicker than % operator)
2448 r = 255;
2449 g = b = 0;
2450 }
2451
2452 // Apply saturation and value to R,G,B, pack into 32-bit result:
2453 uint32_t v1 = 1 + val; // 1 to 256; allows >>8 instead of /255
2454 uint16_t s1 = 1 + sat; // 1 to 256; same reason
2455 uint8_t s2 = 255 - sat; // 255 to 0
2456 return ((((((r * s1) >> 8) + s2) * v1) & 0xff00) << 8) |
2457 (((((g * s1) >> 8) + s2) * v1) & 0xff00) |
2458 ( ((((b * s1) >> 8) + s2) * v1) >> 8);
2459}

2526void Adafruit_NeoPixel::setBrightness(uint8_t b) {
2527 // Stored brightness value is different than what's passed.
2528 // This simplifies the actual scaling math later, allowing a fast
2529 // 8x8-bit multiply and taking the MSB. 'brightness' is a uint8_t,
2530 // adding 1 here may (intentionally) roll over...so 0 = max brightness
2531 // (color values are interpreted literally; no scaling), 1 = min
2532 // brightness (off), 255 = just below max brightness.
2533 uint8_t newBrightness = b + 1;
2534 if(newBrightness != brightness) { // Compare against prior value
2535 // Brightness has changed -- re-scale existing data in RAM,
2536 // This process is potentially "lossy," especially when increasing
2537 // brightness. The tight timing in the WS2811/WS2812 code means there
2538 // aren't enough free cycles to perform this scaling on the fly as data
2539 // is issued. So we make a pass through the existing color data in RAM
2540 // and scale it (subsequent graphics commands also work at this
2541 // brightness level). If there's a significant step up in brightness,
2542 // the limited number of steps (quantization) in the old data will be
2543 // quite visible in the re-scaled version. For a non-destructive
2544 // change, you'll need to re-render the full strip data. C'est la vie.
2545 uint8_t c,
2546 *ptr = pixels,
2547 oldBrightness = brightness - 1; // De-wrap old brightness value
2548 uint16_t scale;
2549 if(oldBrightness == 0) scale = 0; // Avoid /0
2550 else if(b == 255) scale = 65535 / oldBrightness;
2551 else scale = (((uint16_t)newBrightness << 8) - 1) / oldBrightness;
2552 for(uint16_t i=0; i<numBytes; i++) {
2553 c = *ptr;
2554 *ptr++ = (c * scale) >> 8;
2555 }
2556 brightness = newBrightness;
2557 }
