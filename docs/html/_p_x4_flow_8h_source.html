36// As described in the documentation
37// http://pixhawk.org/modules/px4flow
38typedef struct i2c_frame
39{
40 uint16_t frame_count;// counts created I2C frames
41 int16_t pixel_flow_x_sum;// accumulated x flow in pixels*10 since last I2C frame
42 int16_t pixel_flow_y_sum;// accumulated y flow in pixels*10 since last I2C frame
43 int16_t flow_comp_m_x;// x velocity*1000 in meters / timestep
44 int16_t flow_comp_m_y;// y velocity*1000 in meters / timestep
45 int16_t qual;// Optical flow quality / confidence 0: bad, 255: maximum quality
46 int16_t gyro_x_rate; //gyro x rate
47 int16_t gyro_y_rate; //gyro y rate
48 int16_t gyro_z_rate; //gyro z rate
49 uint8_t gyro_range; // gyro range
50 uint8_t sonar_timestamp;// timestep in milliseconds between I2C frames
51 int16_t ground_distance;// Ground distance in meters*1000. Positive value: distance known. Negative value: Unknown distance
52} i2c_frame;
53
54typedef struct i2c_integral_frame
55{
56 uint16_t frame_count_since_last_readout;//number of flow measurements since last I2C readout [#frames]
57 int16_t pixel_flow_x_integral;//accumulated flow in radians*10000 around x axis since last I2C readout [rad*10000]
58 int16_t pixel_flow_y_integral;//accumulated flow in radians*10000 around y axis since last I2C readout [rad*10000]
59 int16_t gyro_x_rate_integral;//accumulated gyro x rates in radians*10000 since last I2C readout [rad*10000]
60 int16_t gyro_y_rate_integral;//accumulated gyro y rates in radians*10000 since last I2C readout [rad*10000]
61 int16_t gyro_z_rate_integral;//accumulated gyro z rates in radians*10000 since last I2C readout [rad*10000]
62 uint32_t integration_timespan;//accumulation timespan in microseconds since last I2C readout [microseconds]
63 uint32_t sonar_timestamp;// time since last sonar update [microseconds]
64 int16_t ground_distance;// Ground distance in meters*1000 [meters*1000]
65 int16_t gyro_temperature;// Temperature * 100 in centi-degrees Celsius [degcelsius*100]
66 uint8_t quality;// averaged quality of accumulated flow values [0:bad quality;255: max quality]
67} i2c_integral_frame;
