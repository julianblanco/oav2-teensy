149float NAVIGATION::crossTrackCorrection(float distanceXT, float targetHead, float distance2WP)
150{
151 float xtCoeff = -100; // based on experimental data from the autonomous car
152 float temp = (xtCoeff * distanceXT) / distance2WP;
153
154 if (temp > 30)
155 temp = 30; // maximum allowable correction
156 if (temp < -30)
157 temp = -30;
158
159 float newTargetHeading = targetHead + temp;
160
161 if (newTargetHeading >= 360)
162 newTargetHeading -= 360;
163 else if (newTargetHeading < 0)
164 newTargetHeading += 360;
165
166 return newTargetHeading;
167} // end crossTrackError

169float NAVIGATION::crossTrackError(float distance2WP, float tracklegHead, float targetHead)
170{
171 //convert to radians for use with sin
172 tracklegHead = (3.14159265 / 180) * tracklegHead;
173 targetHead = (3.14159265 / 180) * targetHead;
174
175 //compute heading error off trackline
176 float deltaHeading = tracklegHead - targetHead;
177
178 // crosstrack distance (positive if right of track)
179 float distanceXT = distance2WP * sin(deltaHeading);
180
181 return distanceXT;
182}

195float NAVIGATION::courseToWaypoint(float lat1, float long1, float lat2, float long2)
196{
197 // returns course in degrees (North=0, West=270) from position 1 to position 2,
198 // both specified as signed decimal-degrees latitude and longitude.
199 // Because Earth is no exact sphere, calculated course may be off by a tiny fraction.
200 // Courtesy of Maarten Lamers
201 float dlon = radians(long2 - long1);
202 lat1 = radians(lat1);
203 lat2 = radians(lat2);
204 float a1 = sin(dlon) * cos(lat2);
205 float a2 = sin(lat1) * cos(lat2) * cos(dlon);
206 a2 = cos(lat1) * sin(lat2) - a2;
207 a2 = atan2(a1, a2);
208 if (a2 < 0.0)
209 {
210 a2 += TWO_PI;
211 }
212 return degrees(a2);
213} // courseToWaypoint()
